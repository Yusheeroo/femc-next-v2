widgets – Композиция фич в блоки

Что хранится:
Крупные составные компоненты, собранные из фич и сущностей.
Header, Footer, Sidebar, Dashboard.

Пример:
import { Logo } from '@/shared/ui';
import { Navigation } from '@/features/navigation';
import { UserAvatar } from '@/entities/user';

export const Header = () => (
  <header>
    <Logo />
    <Navigation />
    <UserAvatar />
  </header>
);
